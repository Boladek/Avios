CREATE TABLE  products_varieties(
    id int AUTO_INCREMENT, 
    size INT,
    color VARCHAR(255), 
    quantity INT,
    images VARCHAR(255), 
    price INT, 
    productId INT, 
    PRIMARY KEY (id), 
    CONSTRAINT FK_ProductsVarieties
    FOREIGN KEY (productId) REFERENCES products(id) ON DELETE CASCADE ON UPDATE CASCADE
    );
